
"DEMO#3: CONTROL DE TRÁFICO: INGRESS GATEWAY": 
---------------------------------------------

kind: Gateway
apiVersion: networking.istio.io/v1beta1
metadata:
  name: my-istio-gateway-demo3
  namespace: ace-istio
spec:
  servers:
    - port:
        number: 80
        protocol: HTTP
        name: http
      hosts:
        - '*'
  selector:
    istio: ingressgateway

--- 

apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: my-istio-service-demo3
  namespace: ace-istio 
spec:
  hosts:
    - istio-ingressgateway-istio-system.apps.tarkin.coc-ibm.com
  gateways:
    - my-istio-gateway-demo3
  http:
    - match:
        - uri:
            prefix: /employeeservice
      route:
        - destination:
            host: dummy-csm-rest-topdown-03a-is
            port:
              number: 7800
